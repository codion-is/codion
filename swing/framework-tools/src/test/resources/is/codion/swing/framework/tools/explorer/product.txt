public interface Product {
  EntityType<Entity> TYPE = DOMAIN.entityType("petstore.product");

  Attribute<Integer> PRODUCTID = TYPE.integerAttribute("productid");
  Attribute<Integer> CATEGORYID = TYPE.integerAttribute("categoryid");
  Attribute<String> NAME = TYPE.stringAttribute("name");
  Attribute<String> DESCRIPTION = TYPE.stringAttribute("description");
  Attribute<String> IMAGEURL = TYPE.stringAttribute("imageurl");

  ForeignKey CATEGORY_FK = TYPE.foreignKey("category_fk", Product.CATEGORYID, Category.CATEGORYID);
}

void product() {
  define(Product.TYPE,
          columnProperty(Product.PRODUCTID)
                .primaryKeyIndex(0),
          columnProperty(Product.CATEGORYID)
                .nullable(false),
          foreignKeyProperty(Product.CATEGORY_FK, "Category"),
          columnProperty(Product.NAME, "Name")
                .nullable(false)
                .maximumLength(25),
          columnProperty(Product.DESCRIPTION, "Description")
                .nullable(false)
                .maximumLength(255),
          columnProperty(Product.IMAGEURL, "Imageurl")
                .maximumLength(55)
  );
}
