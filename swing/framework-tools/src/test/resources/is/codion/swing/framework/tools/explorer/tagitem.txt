public interface TagItem {
  EntityType TYPE = DOMAIN.entityType("petstore.tag_item");

  Attribute<Integer> TAGID = TYPE.integerAttribute("tagid");
  Attribute<Integer> ITEMID = TYPE.integerAttribute("itemid");

  ForeignKey TAGID_FK = TYPE.foreignKey("tagid_fk", TagItem.TAGID, Tag.TAGID);
  ForeignKey ITEMID_FK = TYPE.foreignKey("itemid_fk", TagItem.ITEMID, Item.ITEMID);
}

void tagItem() {
  define(TagItem.TYPE,
          columnProperty(TagItem.TAGID)
                .primaryKeyIndex(0),
          foreignKeyProperty(TagItem.TAGID_FK, "Tag"),
          columnProperty(TagItem.ITEMID)
                .primaryKeyIndex(1),
          foreignKeyProperty(TagItem.ITEMID_FK, "Item")
  );
}
