public interface Address {
  EntityType<Entity> TYPE = DOMAIN.entityType("petstore.address");
  Attribute<Integer> ADDRESSID = TYPE.integerAttribute("addressid");
  Attribute<String> STREET1 = TYPE.stringAttribute("street1");
  Attribute<String> STREET2 = TYPE.stringAttribute("street2");
  Attribute<String> CITY = TYPE.stringAttribute("city");
  Attribute<String> STATE = TYPE.stringAttribute("state");
  Attribute<Integer> ZIP = TYPE.integerAttribute("zip");
  Attribute<Double> LATITUDE = TYPE.doubleAttribute("latitude");
  Attribute<Double> LONGITUDE = TYPE.doubleAttribute("longitude");
}

void address() {
  define(Address.TYPE,
          columnProperty(Address.ADDRESSID)
                .primaryKeyIndex(0),
          columnProperty(Address.STREET1, "Street1")
                .nullable(false)
                .maximumLength(55),
          columnProperty(Address.STREET2, "Street2")
                .maximumLength(55),
          columnProperty(Address.CITY, "City")
                .nullable(false)
                .maximumLength(55),
          columnProperty(Address.STATE, "State")
                .nullable(false)
                .maximumLength(25),
          columnProperty(Address.ZIP, "Zip")
                .nullable(false),
          columnProperty(Address.LATITUDE, "Latitude")
                .nullable(false)
                .maximumFractionDigits(2),
          columnProperty(Address.LONGITUDE, "Longitude")
                .nullable(false)
                .maximumFractionDigits(2)
  );
}
